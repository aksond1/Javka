public class SumOddRange {
    // write code here
    public static boolean isOdd(int number) {
        if (number > 0) {
            if (number % 2 == 1) {
                return true;
            }
        }
        return false;
    }
    
    public static int sumOdd(int start, int end) {
        int sum = 0;
        
        for (int i = start; i <= end; i++) {
            if (isOdd(i)) {
                sum += i;
            }
        }
        
        if (end < start || start < 0 || end < 0) {
            return -1;
        }
        return sum;
    }
}
